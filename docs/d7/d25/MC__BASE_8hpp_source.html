234  boost::random::random_device rd;
235  boost::random::uniform_real_distribution<> uniform(0.0, 1.0);
236  boost::random::mt19937 generator(rd);
237 
238  std::vector<int> current_loc(size, 0);
239  // run monte calro sampling using dagger sampling scheme
240  for (unsigned int i = 1; i <= num; ++i)
241  {
242  if (i_DB)
243  {
244  // update MC fragility
245  i_DB->update_fragility(i);
246  }
247 
248  // update BE state
249  for (unsigned int j = 0; j < size; ++j)
250  {
251  if (!i_DB)
252  {
253  if (loc[j][current_loc[j]] == (int)(i))
254  {
255  ++current_loc[j];
256  boost::static_pointer_cast<FT::BE>(BEs[j])->assign_prob(1.0);
257  }
258  else
259  {
260  boost::static_pointer_cast<FT::BE>(BEs[j])->assign_prob(0.0);
261  }
262  }
263  else
264  {
265  if (boost::static_pointer_cast<FT::BE>(BEs[j])->get_prob() >= uniform(generator))
266  {
267  boost::static_pointer_cast<FT::BE>(BEs[j])->assign_prob(1.0);
268  }
269  else
270  {
271  boost::static_pointer_cast<FT::BE>(BEs[j])->assign_prob(0.0);
272  }
273  }
274 
275  }
276

278 
279  _mean += tmp;
280  _var += tmp*tmp;
281

362  {

365  // initialize random number generator
366  std::random_device rd;
367 
368  //default engine
369  std::default_random_engine engine(rd());
370 
371  std::vector<int> sequence(num);
372  std::iota(std::begin(sequence), std::end(sequence), 1);
373 
374  std::vector<std::vector<int>> loc;
375  // initialize dagger sampling
376  for (unsigned int i = 0; i < size; ++i)
377  {
378  // shuffle sequence
379  std::shuffle(std::begin(sequence), std::end(sequence), engine);
380 
381  // @todo this approach is inefficient for a basic event close to 1.
382 
383  // expected number of failures
384  int num_fail = int(floor(probs[i] * num));
385 
386  // quotient
387  if (double(probs[i] * num) != floor(probs[i] * num))
388  ++num_fail;
389 
390  std::vector<int> tmp;
391 
392  for (auto j = 0; j < num_fail; ++j)
393  tmp.push_back(sequence[j]);
394 
395  // order it
396  std::sort(tmp.begin(), tmp.end());
397 
398  // add dummy number
399  tmp.push_back(num + 1);
400 
401  loc.push_back(tmp);
402  }
403  return(loc);
