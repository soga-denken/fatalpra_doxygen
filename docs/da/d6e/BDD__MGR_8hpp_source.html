252  template<class _BDD, class _MGR>
253  _BDD bdd_kn(const int & k, typename std::vector<_BDD>::iterator begin, const int & i, const boost::shared_ptr<FATAL::DD::dd_mgr<_MGR, _BDD>> & i_mgr)
254  {
255  // k = 0
256  if (k == 0)
257  return i_mgr->One();
258 
259  // check if sufficient elements remain to satisfy KN condition
260  if (i < k)
261  return i_mgr->Zero();
262 
263  // now, i >= k
264  _BDD val = *begin;
265 
266  ++begin;
267 
268  _BDD success = bdd_kn(k - 1, begin, i - 1, i_mgr) * val;
269 
270  _BDD failure = bdd_kn(k, begin, i - 1, i_mgr) * (!val);
271 
272  return(success + failure);
