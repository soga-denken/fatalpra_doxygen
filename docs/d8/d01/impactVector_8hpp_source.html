412  // mapping up factor estimation method 1
413  // F.M. Marshall et. al., Common-Cause Failure Database andAnalysis System: Event De$nition and Classification, NUREGICR-6268, Volume 2, INEELAZXT-97-00696,, Washington, D.C. (1998).
414  // Kvam, P., "Estimation Techniques for Common-cause Failure Data with Different System Sizes," Technometrics, 38(4):382-388, 1996.
415  // quote from NUREG/CR-XXX
416  /* "This method works well when the system sizes are close to one another (e.g., mapping from size 2 to size 3 or 4) or when
417  at least one of the component degradation values is less than 1.0. When all of the component degradation values are equal
418  to 1.0, p is also equal to 1.0. When used in the mapping up equations for the RPS data, this method tends to overestimate
419  the 'probability that components added to a system will exhibit the same lethal shock-like behavior. "*/
420  const double rho_NUREGCR_6268(const std::vector<double> & F)
421  {
422  auto size = F.size() - 1;
423 
424  //numerator
425  double numerator = 0;
426 
427  // the summation of the original equation starts from 1, but 2 is fine.
428  for (size_t i = 2; i <= size; ++i)
429  {
430  numerator += i * (i - 1) * F[i];
431  }
432 
433  //denomerator
434  double denomerator = 0;
435 
436  // memo: summation starts from 1 to m.
437  for (size_t i = 1; i <= size; ++i)
438  {
439  denomerator += i * F[i];
440  }
441 
442  return numerator / denomerator / (double(size) - 1.0);
443  }
